# _.flow

_.flow([funcs])创建一个函数。 返回的结果是调用提供函数的结果，this 会绑定到创建函数。 每一个连续调用，传入的参数都是前一个函数返回的结果。

传入函数的数组[funcion,funcion,...funcions]创建函数(简称 flow)。每次调用flow(...arg)，将参数...arg传入第一个 function，返回的结果作为参数传入第二个 function，如此类推，最后返回结果。

源码：

function flow(...funcs) {
    const length = funcs.length;
    let index = length;
    //检查有没有非function参数
    while (index--) {
      if (typeof funcs[index] !== 'function') {
        throw new TypeError('Expected a function');
      }
    }
    //初始化
    return function (...args) {
      let index = 0;
      //...args  参数解构得出解构是数组 所以用apply
      //.apply(this,[ arg0,arg1,arg2])
      let result = length ? funcs[index].apply(this, args) : args[0];
      //循环方式执行
      while (++index < length) {
        //每次循环把result当参数输入，得出新的result就赋值
        //.call(this, arg0,arg1,arg2)
        result = funcs[index].call(this, result);
      }
      //循环完毕返回result
      return result;
    };

function compose() {
  //没有检查类型   顺序是倒过来的，但实现方式都一样
  var args = arguments;
  var start = args.length - 1;
  return function () {
    var i = start;
    var result = args[start].apply(this, arguments);
    while (i--) result = args[i].call(this, result);
    return result;
  };
}

function compose(...fns) {
  return function composed(result) {
    // 拷贝一份保存函数的数组
    var list = fns.slice();
    while (list.length > 0) {
      // 将最后一个函数从列表尾部拿出
      // 并执行它
      result = list.pop()(result);
    }
    return result;
  };
}
